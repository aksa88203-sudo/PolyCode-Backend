void ShoppingCart::removeItem(int id)
{
    if (!head)
    {
        cout << "Cart is empty!" << endl;
        return;
    }
    if (head->id == id)
    {
        CartItem *tmp = head;
        head = head->next;
        delete tmp;
        size--;
        cout << "Item removed successfully!" << endl;
        return;
    }
    CartItem *cur = head;
    while (cur->next && cur->next->id != id)
    {
        cur = cur->next;
    }
    if (!cur->next)
    {
        cout << "Item not found!" << endl;
        return;
    }
    CartItem *tmp = cur->next;
    cur->next = tmp->next;
    delete tmp;
    size--;
    cout << "Item removed successfully!" << endl;
}

bool ShoppingCart::isNumeric(const string &s)
{
    if (s.empty())
    {
        return false;
    }
    int start = 0;
    if (s[0] == '-')
    {
        start = 1;
    }
    bool hasDot = false;
    for (int i = start; i < (int)s.size(); i++)
    {
        if (s[i] == '.' && !hasDot)
        {
            hasDot = true;
            continue;
        }
        if (s[i] < '0' || s[i] > '9')
        {
            return false;
        }
    }

}

### 🔄 Execution Flow Examples

#### Adding an Item with Attributes:
1. `addItem(1, "Laptop")` called
2. `findItem(1)` returns `nullptr` (item doesn't exist)
3. `new CartItem(1, "Laptop")` allocates memory
4. CartItem constructor initializes `id=1`, `name="Laptop"`, `attrs=nullptr`, `next=nullptr`
5. Item appended to cart list
6. `addAttribute(1, "price", "999.99")` called
7. `findItem(1)` returns pointer to Laptop item
8. `new Attribute("price", "999.99")` allocates memory
9. Attribute linked to item's attribute list

#### Sorting Process:
1. `sortCartByAttr("price")` called
2. Array of pointers created: `[item1_ptr, item2_ptr, item3_ptr]`
3. Bubble sort compares `getAttrValue()` for each adjacent pair
4. Pointers swapped based on numeric comparison
5. Linked list rebuilt from sorted pointer array
6. Temporary array freed

### 🎯 Best Practices Demonstrated

1. **Resource Acquisition Is Initialization (RAII)**
   - Constructors initialize objects to valid state
   - Destructors automatically clean up resources

2. **Defensive Programming**
   - Null pointer checks before dereferencing
   - Input validation for edge cases
   - Early returns for error conditions

3. **Efficient Memory Usage**
   - Pass by reference to avoid copies
   - Minimal temporary allocations
   - Proper cleanup in all code paths

4. **Clear Interface Design**
   - Public methods provide high-level operations
   - Private helpers hide implementation details
   - Consistent naming conventions

5. **Separation of Concerns**
   - Data structures separate from business logic
   - Input/output separated from core functionality
   - Each method has single responsibility
